Interface YahtzeeState

Represents the complete state of a Yahtzee game. Tracks all game-related information including scores, dice, and turn status.

Example

const gameState: YahtzeeState = {
  scores: {
    "player1": { ones: 3, twos: 8, total: 11 },
    "player2": { ones: 4, twos: 6, total: 10 }
  },
  dice: [1, 3, 3, 4, 6],
  heldDice: [false, true, true, false, false],
  rollsLeft: 2,
  currentPlayer: "player1",
  gameOver: false,
  turnStarted: true,
  round: 1
};
